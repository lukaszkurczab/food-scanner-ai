# Reference rules snapshot for Fitaly Firebase project.
# Keep this file aligned with deployed Firebase Storage and Firestore rules.

# Storage

rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    function isAuthenticated() {
      return request.auth != null;
    }

    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }

    function isAdmin() {
      return isAuthenticated() && request.auth.token.admin == true;
    }

    match /avatars/{userId}/{filename} {
      allow read, write: if isOwner(userId)
        && filename.matches('^avatar(\\.\\w+)?$');
    }

    match /meals/{userId}/{filename} {
      allow read, write: if isOwner(userId);
    }

    match /myMeals/{userId}/{filename} {
      allow read, write: if isOwner(userId);
    }

    match /feedback/{userId}/{feedbackId}/{filename} {
      allow write: if isOwner(userId);
      allow read: if isOwner(userId) || isAdmin();
    }

    // Legacy storage path; sunset target: 2026-06-30.
    match /feedbacks/{document=**} {
      allow read, write: if false;
    }

    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}

# Firestore database

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function isAuthenticated() {
      return request.auth != null;
    }

    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }

    function isAdmin() {
      return isAuthenticated() && request.auth.token.admin == true;
    }

    function isValidStreak() {
      return request.resource.data.keys().hasOnly(['current', 'lastDate'])
          && request.resource.data.current is int
          && request.resource.data.current >= 0
          && (
               request.resource.data.lastDate == null ||
               (request.resource.data.lastDate is string &&
                request.resource.data.lastDate.matches('^\\d{4}-\\d{2}-\\d{2}$'))
             );
    }

    match /usernames/{username} {
      // Public read for username availability checks.
      allow get: if true;
      allow list: if false;

      allow create: if isAuthenticated()
                    && request.resource.data.uid is string
                    && request.resource.data.uid == request.auth.uid;

      allow update: if isAuthenticated()
                    && resource.data.uid is string
                    && request.resource.data.uid is string
                    && resource.data.uid == request.auth.uid
                    && request.resource.data.uid == request.auth.uid;

      allow delete: if isAuthenticated()
                    && resource.data.uid is string
                    && resource.data.uid == request.auth.uid;
    }

    match /users/{userId} {
      allow create: if isOwner(userId);
      allow read, update, delete: if isOwner(userId);
    }

    match /users/{userId}/meals/{mealId} {
      allow create, read, update, delete: if isOwner(userId);
    }

    match /users/{userId}/myMeals/{mealId} {
      allow create, read, update, delete: if isOwner(userId);
    }

    match /users/{userId}/chat_threads/{threadId} {
      allow read, create, update, delete: if isOwner(userId);

      match /messages/{messageId} {
        allow read, create, update, delete: if isOwner(userId);
      }

      // Backend-owned internal memory state.
      match /memory/{memoryId} {
        allow read, write: if false;
      }
    }

    match /users/{userId}/streak/{docId} {
      allow read, delete: if isOwner(userId);
      allow create, update: if isOwner(userId) && isValidStreak();
    }

    // Backend-owned write model; client reads only.
    match /users/{userId}/badges/{badgeId} {
      allow read: if isOwner(userId);
      allow write: if false;
    }

    // Backend-owned write model; client reads and can mark as deleted.
    match /users/{userId}/notifications/{notifId} {
      allow read, delete: if isOwner(userId);
      allow create, update: if false;
    }

    match /users/{userId}/prefs/{docId} {
      allow read, create, update, delete: if isOwner(userId);
    }

    // Backend writes feedback docs; client may only read/create own entries.
    match /users/{userId}/feedback/{feedbackId} {
      allow create, read: if isOwner(userId);
      allow update, delete: if false;
    }

    // Backend-owned reminder state; not exposed to client SDK.
    match /users/{userId}/daily_stats/{docId} {
      allow read, write: if false;
    }

    // Billing snapshot + ledger (user-owned path, backend-owned writes).
    match /users/{userId}/billing/main/aiCredits/current {
      allow read: if isOwner(userId);
      allow write: if false;
    }

    match /users/{userId}/billing/main/aiCreditTransactions/{txId} {
      allow read: if isOwner(userId);
      allow write: if false;
    }

    // Any other billing document path is rejected by default.
    match /users/{userId}/billing/{document=**} {
      allow read, write: if false;
    }

    // Legacy meal mirror; sunset target: 2026-06-30.
    match /users/{userId}/history/{mealId} {
      allow read, write: if false;
    }

    // Legacy top-level Firestore paths to decommission after cutover.
    match /feedbacks/{document=**} {
      allow read, write: if false;
    }

    match /ai_credits/{document=**} {
      allow read, write: if false;
    }

    match /ai_credit_transactions/{document=**} {
      allow read, write: if false;
    }

    // AI gateway diagnostics no longer persist in Firestore.
    match /ai_gateway_logs/{document=**} {
      allow read, write: if false;
    }

    // Telemetry / operational stores are backend-managed only.
    match /telemetry_events/{document=**} {
      allow read, write: if false;
    }

    match /ai_runs/{document=**} {
      allow read, write: if false;
    }

    match /rate_limits/{document=**} {
      allow read, write: if false;
    }

    match /_healthcheck/{document=**} {
      allow read, write: if false;
    }

    // Explicit deny-by-default fallback.
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
